navigation

bi.navigation

导航栏控件,BI.Widget

source



BI.createWidget({
  element: "#wrapper",
  type: "bi.navigation",
  tab: {
    height: 30,
    items: [{
      once: false,
      text: "后退",
      value: -1
    }, {
      once: false,
      text: "前进",
      value: 1
    }]
  },
  cardCreator: function(v) {
    return BI.createWidget({
      type: "bi.label",
      cls: "layout-bg" + BI.random(1, 8),
      text: "第" + v + "页"
    })
  }
})
